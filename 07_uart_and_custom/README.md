```c
printf("\r\nUART example started.\r\n");

while (true)
{
    uint8_t cr;
    while (app_uart_get(&cr) != NRF_SUCCESS);
    while (app_uart_put(cr) != NRF_SUCCESS);

    if (cr == '\n' || cr == '\r')
    {
        printf("\r\n>>");
    }
}
```
